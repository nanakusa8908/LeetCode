# 2542. Maximum Subsequence Score
###### tags: `LeetCode`
## **Link**
https://leetcode.com/problems/maximum-subsequence-score/
## **Code**
```cpp=
class Solution {
public:
    long long maxScore(vector<int>& nums1, vector<int>& nums2, int k) 
    {
        int n=nums1.size();
        priority_queue<int,vector<int>,greater<int>> pq;
        long long ans=0,now=0,ct=0,mn=99999999;
        vector<pair<long long,long long>> vt;
        for(int i=0;i<n;++i)
            vt.push_back(make_pair(nums2[i],nums1[i]));
        sort(vt.begin(),vt.end());
        for(int i=n-1;i>=0;--i)
        {
            if(ct<k-1)
            {
                now+=vt[i].second;
                mn=min(mn,vt[i].first);
                pq.push(vt[i].second);
                ++ct;
            }
            else
            {
                mn=min(mn,vt[i].first);
                ans=max(ans,(now+vt[i].second)*mn);
                if(!pq.empty() && pq.top()<vt[i].second)
                {
                    now-=pq.top();
                    now+=vt[i].second;
                    pq.pop();
                    pq.push(vt[i].second);
                }
            }
        }
        return ans;
    }
};
```
