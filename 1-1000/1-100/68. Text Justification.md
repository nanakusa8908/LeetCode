# 68. Text Justification
###### tags: `LeetCode`
## **Link**
https://leetcode.com/problems/text-justification/description/
## **Code**
```cpp=
class Solution {
public:
    vector<string> fullJustify(vector<string>& words, int maxWidth) 
    {
        vector<int> sz;
        for(string &v:words)
            sz.push_back(v.size());
        vector<string> ans;
        string st;
        int i,now=0,cnt,spc,mod;
        while(now<words.size())
        {
            cnt=0;
            //看這節可以塞多少單字
            for(i=now;i<words.size();++i)
            {
                if(cnt+sz[i]+i-now<=maxWidth)
                    cnt+=sz[i];
                else
                    break;
            }
            --i;
            st.clear();
            //情況一：最後一行
            if(i+1==words.size())
            {
                for(int j=now;j<=i;++j)
                {
                    st+=words[j];
                    if(i-j)
                        st+=string(1,' ');
                }
                st+=string(maxWidth-st.size(),' ');
            }
            //情況二：非情況一且一行大於1個單字
            else if(i-now)
            {
                spc=(maxWidth-cnt)/(i-now);
                mod=(maxWidth-cnt)%(i-now);
                for(int j=now;j<=i;++j)
                    if(i-j)
                        st+=words[j]+string(spc+(mod>j-now),' ');
                    else
                        st+=words[j];
            }
            //情況三：非情況一且非情況二
            else
            {
                spc=maxWidth-cnt;
                mod=0;
                st=words[i]+string(spc,' ');
            }
            ans.push_back(st);
            now=++i;
        }
        return ans;
    }
};
```
