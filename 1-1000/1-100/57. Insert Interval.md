# 57. Insert Interval
###### tags: `LeetCode`
## **Link**
https://leetcode.com/problems/insert-interval/

## **Code**
```cpp=
class Solution {
public:
    vector<vector<int>> insert(vector<vector<int>>& intervals, vector<int>& newInterval) 
    {
        vector<vector<int>> ans;
        vector<int> ins(2);
        bool fg=false;
        for(int i=0;i<intervals.size();++i)
        {
            if(!fg && intervals[i][1]>=newInterval[0]) //進到合併範圍
            {
                if(intervals[i][0]>=newInterval[0]) //判斷合併後的頭是誰
                {
                    ins[0]=newInterval[0];
                }
                else
                {
                    ins[0]=intervals[i][0];
                }
                do //往後找合併的末端在哪裡，因為自己也要檢查，所以我用do while
                {
                    if(intervals[i][1]>=newInterval[1]) //到合併末端
                    {
                        if(intervals[i][0]>newInterval[1]) //判斷合併後的尾是誰
                        {
                            ins[1]=newInterval[1];
                            --i;
                        }
                        else
                        {
                            ins[1]=intervals[i][1];
                        }
                        fg=true;
                        ans.push_back(ins);
                        break;
                    }
                }while(++i<intervals.size());
                if(!fg) //特判1：找到底還沒有找到末端 => newInterval就是末端
                {
                    fg=true;
                    ins[1]=newInterval[1];
                    ans.push_back(ins);
                }
            }
            else
            {
                ans.push_back(intervals[i]);
            }
        }
        if(!fg) //特判2：完全沒有做過合併 => 空interval 或是 newInterval在interval的後方
        {
            ans.push_back(newInterval);
        }
        return ans;
    }
};
```
