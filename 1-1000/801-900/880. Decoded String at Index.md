# 880. Decoded String at Index
###### tags: `LeetCode`
## **Link**
https://leetcode.com/problems/decoded-string-at-index
## **Code**
```cpp=
class Solution {
public:
    string decodeAtIndex(string s, int k) 
    {
        long long sz=0,i;
        // 找到包含k的最小長度
        // 其實可以直接把decode後的長度算出來 題目保證不會溢位
        for(i=0;sz<k && i<s.size();++i)
        {
            // 算目前長度
            if(isdigit(s[i]))
                sz*=(s[i]-'0');
            else
                ++sz;
        }
        --i;
        for(;i>=0;--i)
        {
            // 找到了
            if(!isdigit(s[i]) && (k==0 || sz==k))
                break;
            // 扣長度
            if(isdigit(s[i]))
            {
                sz/=(s[i]-'0');
                k%=sz;
            }
            else
                --sz;
        }
        // 題目要求用字串回傳 =w=
        // return string(1,s[i]);
        // 也可以
        string st;
        st+=s[i];
        return st;
    }
};
```
