# 491. Non-decreasing Subsequences
###### tags: `LeetCode`
## **Link**
https://leetcode.com/problems/non-decreasing-subsequences/
## **迴圈解**
```cpp=
class Solution {
public:
    vector<vector<int>> findSubsequences(vector<int>& nums) 
    {
        int n=nums.size();
        vector<int> tmp;
        vector<vector<int>> ans;
        vector<set<vector<int>>> kp(n); // 存以nums[i]結尾的所有可行陣列
        set<vector<int>> check; // 處理重複的答案
        for(int i=0;i<n;++i)
        {
            for(int j=0;j<i;++j)
            {
                if(nums[j]<=nums[i])
                {
                    for(auto v:kp[j])
                    {
                        tmp=v;
                        tmp.push_back(nums[i]);
                        kp[i].insert(tmp);
                        check.insert(tmp);
                    }
                    kp[i].insert({nums[j],nums[i]});
                    check.insert({nums[j],nums[i]});
                }
            }
        }
        for(auto v:check)
            ans.push_back(v);
        return ans;
    }
};  
```
## **超慢遞迴解**
```cpp=
class Solution {
public:
    void dfs(int n, set<vector<int>> &st, vector<int> &keep, vector<int> &nums)
    {
        if(n==nums.size())  
            return;
        for(int i=n;i<nums.size();++i)
        {
            if(keep.empty())
                keep.push_back(nums[i]);
            else
                if(keep.back()<=nums[i])
                    keep.push_back(nums[i]);
                else
                    return;
            if(keep.size()>1)
                st.insert(keep);
            for(int j=i+1;j<nums.size();++j)
                dfs(j,st,keep,nums);
            keep.pop_back();
        }
    }

    vector<vector<int>> findSubsequences(vector<int>& nums) 
    {
        vector<vector<int>> ans;
        set<vector<int>> st;
        vector<int> keep;
        dfs(0,st,keep,nums);
        for(auto v:st)
            ans.push_back(v);
        return ans;
    }
};
```
