# 1156. Swap For Longest Repeated Character Substring
###### tags: `LeetCode`
## **Link**
https://leetcode.com/problems/swap-for-longest-repeated-character-substring/

## **Code**
```cpp=
class Solution { 
public:
    int maxRepOpt1(string text) 
    {
        vector<pair<char,int>> vt;
        char ch=text[0];
        int ct=0;
        for(char v:text)
        {
            if(ch==v)
                ++ct;
            else
            {
                vt.push_back(make_pair(ch,ct));
                ch=v;
                ct=1;
            }
        }
        vt.push_back(make_pair(ch,ct));
        if(vt.size()==1)
            return vt[0].second;
        else if(vt.size()==2)
            return max(vt[0].second,vt[1].second);
        else
        {
            int ans=-1;
            for(pair<char,int> v:vt)
                ans=max(ans,v.second);
            for(int i=0;i<vt.size();++i)
            {
                if(i+2<vt.size() && vt[i].first==vt[i+2].first && vt[i+1].second==1)
                {
                    for(int j=0;j<vt.size();++j)
                    {
                        if(j!=i && j!=i+2 && vt[j].first==vt[i].first)
                        {
                            ans=max(ans,vt[i].second+vt[i+2].second+1);
                            break;
                        }
                        if(j+1==vt.size())
                            ans=max(ans,vt[i].second+vt[i+2].second);
                    }
                }
                else
                {
                    for(int j=0;j<vt.size();++j)
                    {
                        if(j!=i && vt[j].first==vt[i].first)
                        {
                            ans=max(ans,vt[i].second+1);
                            break;
                        }
                    }
                }
            }
            return ans;
        }
        
    }
};
```
