# 1048. Longest String Chain
###### tags: `LeetCode`

https://leetcode.com/problems/longest-string-chain/
## **Code**

class Solution {
public:
    int longestStrChain(vector<string>& words) 
    {
        ios::sync_with_stdio(false);
        cin.tie(0);
        cout.tie(0);
        vt.resize(words.size(),0);
        // 按照字串長度排序
        sort(words.begin(),words.end(),[](string &a, string &b){return a.size()<b.size();});
        // 從最長的字串開始 如果這個字串還沒找過的話就找他可以接多長
        for(int i=words.size()-1;i>=0;--i)
            if(vt[i]==0)
                vt[i]=solve(words,i);
        int ans=0;
        for(int &v:vt)
            ans=max(ans,v);
        return ans;
    }
private:
    vector<int> vt;
    int solve(vector<string> &words, int idx)
    {
        // 已經找過就不用往下找了
        if(t[idx]!=0)
            return vt[idx];
        int ans=0;
        for(int i=idx-1;i>=0;--i)
        {
            // 如果這個字串可以往下接 就去找這個字串可以接多長
            if(check(words[i],words[idx]))
                ans=max(ans,solve(words,i));
        }
        return vt[idx]=ans+1;
    }
    bool check(string &n, string &m)
    {
        if(n.size()!=m.size()-1)
            return false;
        int idx=0;
        for(int i=0;i<m.size();++i)
            if(idx!=n.size() && n[idx]==m[i])
                ++idx;
        return idx==n.size();
    }
};
