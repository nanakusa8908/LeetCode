# 1962. Remove Stones to Minimize the Total
###### tags: `LeetCode`
## **Link**
https://leetcode.com/problems/remove-stones-to-minimize-the-total/
## **Code**
```cpp=
class Solution {
public:
    int minStoneSum(vector<int>& piles, int k) 
    {
        priority_queue<int> a;
        int ans=0,now;
        for(int v:piles)
            a.push(v);
        while(k--)
        {
            now=a.top();
            a.pop();
            now-=(now/2);
            a.push(now);
        }
        while(!a.empty())
        {
            ans+=a.top();
            a.pop();
        }
        return ans;
    }
};
```
